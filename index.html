Google Sheets Integration

Send every registration directly to a live Google Sheet in real time.

Step 1 — Create your Google Sheet

Open Google Sheets, create a new spreadsheet named GJMC Member Records. Leave the first row blank — the script creates headers automatically.

Step 2 — Open Apps Script & paste code

Click Extensions → Apps Script. Delete existing code, paste the script below, then save.

function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var headers = ['ID','Timestamp','Member Type','Title','First Name', 'Middle Name','Surname','DOB','Gender','Marital','Nationality', 'Occupation','Phone','Phone 2','Email','Place','House No', 'Landmark','Region','GPS Address','Ministry','Referral', 'Next of Kin','Kin Relation','Kin Phone','Salvation Date', 'Baptism','Prev Church','Prayer','Has Photo']; if (sheet.getLastRow() === 0) { sheet.appendRow(headers); sheet.getRange(1,1,1,headers.length) .setFontWeight('bold') .setBackground('#09111e') .setFontColor('#c9993a'); } sheet.appendRow([data.id,data.timestamp,data.member_type, data.title,data.firstname,data.middlename,data.surname, data.dob,data.gender,data.marital,data.nationality, data.occupation,data.phone,data.phone2,data.email, data.place_live,data.house_number,data.landmark, data.region,data.digital_address,data.ministry, data.referral,data.kin_name,data.kin_rel,data.kin_phone, data.salvation_date,data.baptism,data.prev_church, data.prayer,data.photo?'Yes':'No']); return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error',msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
