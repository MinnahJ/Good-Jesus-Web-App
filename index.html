Google Sheets Integration

Send every registration directly to a live Google Sheet in real time.

Step 1 — Create your Google Sheet

Open Google Sheets, create a new spreadsheet, and name it GJMC Member Records. Leave the first row blank — the script will create headers automatically.

Step 2 — Open Apps Script

In your sheet, click Extensions → Apps Script. Delete any existing code, then paste in the script below and click Save.

function doPost(e) { try { var sheet = SpreadsheetApp .getActiveSpreadsheet() .getActiveSheet(); var data = JSON.parse(e.postData.contents); var headers = [ 'ID','Timestamp','Member Type','Title', 'First Name','Middle Name','Surname','Date of Birth', 'Gender','Marital Status','Nationality','Occupation', 'Salvation Date','Baptism','Prev Church','Phone', 'Phone 2','Email','Area','Region','Digital Address', 'Next of Kin','Kin Relation','Kin Phone', 'Ministry','Referral','Prayer Request','Has Photo' ]; if(sheet.getLastRow() === 0) { sheet.appendRow(headers); sheet.getRange(1,1,1,headers.length) .setFontWeight('bold') .setBackground('#09111e') .setFontColor('#c9993a'); } sheet.appendRow([ data.id, data.timestamp, data.member_type, data.title, data.firstname, data.middlename, data.surname, data.dob, data.gender, data.marital, data.nationality, data.occupation, data.salvation_date, data.baptism, data.prev_church, data.phone, data.phone2, data.email, data.area, data.region, data.digital_address, data.kin_name, data.kin_rel, data.kin_phone, data.ministry, data.referral, data.prayer, data.photo ? 'Yes' : 'No' ]); return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error',message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
Step 3 — Deploy as Web App

Click Deploy → New Deployment. Select type: Web App. Set Execute as: Me and Who has access: Anyone. Click Deploy and copy the Web App URL.
